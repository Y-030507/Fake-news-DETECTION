def wordopt(text):
    text=text.lower()

    text=re.sub(r'https?://\S+|www\.\S+','',text)

    text=re.sub(r'<.*?>','',text)

    text=re.sub(r'[^\w\s]','',text)

    text=re.sub(r'\d','',text)

    text=re.sub(r'\n','',text)
    return text
